function reorderFrom(arr, from) {  
    const a1 = arr.slice(0, from).reverse()  
    const a2 = arr.slice(from);  
    const reordered = []  
    for (let i = 0, l = Math.max(a1.length, a2.length); i < l; i++) {  
        const v2 = a2[i]  
        v2 !== undefined && reordered.push(v2)  
        const v1 = a1[i]  
        v1 !== undefined && reordered.push(v1)  
    }  
    return reordered  
}
